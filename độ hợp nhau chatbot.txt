function boDau(str) {
    return str.normalize("NFD")
              .replace(/[\u0300-\u036f]/g, "")
              .replace(/\s+/g, "")
              .toUpperCase();
}

function demChu(text) {
    let seen = [];
    let counts = [];
    for (let c of text) {
        if (!seen.includes(c)) {
            seen.push(c);
            counts.push(text.split(c).length - 1);
        }
    }
    return counts;
}

function tachSo(n) {
    return n.toString().split("").map(Number);
}

function congDaySo(nums) {
    let digits = [];
    nums.forEach(n => digits.push(...tachSo(n)));

    while (digits.length > 2) {
        let newDigits = [];
        let i = 0, j = digits.length - 1;

        while (i < j) {
            let s = digits[i] + digits[j];
            newDigits.push(...tachSo(s));
            i++; j--;
        }
        if (i === j) newDigits.push(digits[i]);

        digits = newDigits;
    }
    return digits;
}
